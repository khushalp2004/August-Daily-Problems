/////498. Diagonal Traverse

//method- map is used to store diagonal elements and then use toggle flip variable which flip alternately and reverse the key's mapped value and store it in result vector and return result;

vector<int> findDiagonalOrder(vector<vector<int>>& mat) {
        int m=mat.size();
        int n=mat[0].size();

        map<int,vector<int>> mp;
        vector<int>result;
        //fil the map using [i+j]
        for(int i=0;i<m;i++){
            for(int j=0;j<n;j++){
                mp[i+j].push_back(mat[i][j]);
            }
        }

        bool flip=true;
        for(auto &it: mp){
            if(flip){
                reverse(it.second.begin(),it.second.end());
            }

            for(int &num: it.second){
                result.push_back(num);
            }

            flip=!flip;//toggle (directions of arrow)
        }
        return result;
    }
