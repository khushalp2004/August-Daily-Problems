
/////342. Power of Four

//METHOD-1
bool isPowerOfFour(int n) {
        if (n<=0) return 0;
        int exp=31-countl_zero((unsigned)n);// log2(n)
        return (exp&1)==0 && n==1<<exp;
    }


//METHOD-2
bool isPowerOfFour(int n) {
        if (n<0) return 0;
        bitset<31> x(n);
        if (x.count()!=1) return 0;
        for(int i=0; i<31; i+=2)
            if(x[i]) return 1;
        return 0;
    }
