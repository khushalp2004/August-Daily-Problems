////869. Reordered Power of 2

bool reorderedPowerOf2(int n) {
        string s = to_string(n);

        sort(s.begin(),s.end());

        for(int i=0;i<=60;i++){
            long long p= 1LL<<i;
            string t=to_string(p);

            if(t.size()!=s.size()){
                continue;
            }

            sort(t.begin(),t.end());
            if(t==s){
                return true;
            }
        }

        return false;
    }
